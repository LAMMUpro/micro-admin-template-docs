## 注册组件
### vue3
```tsx
defineComponent({
  setup() {
    return () =>
      h(MicroComponent, {
        _is: 'Page404',
        msg: '组件参数',
      });
  },
})
```

### vue2
```tsx
{
  render(createElement) {
    return createElement(MicroComponent, {
      props: {
        _is: 'Page404',
      },
      /** 如果需要传组件参数需要通过attrs传递 */
      attrs: {
        msg: '组件参数',
      },
    });
  },
}
```
